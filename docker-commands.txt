install docker
install docker-compose

create folder for mysql storage
sudo chmod -R 777 <folder>

run commands in project dir:
    docker-compose build app
    docker-compose up -d
    docker-compose exec app composer update
    set mysql user permissions:
        docker exec -it notebook-db /bin/bash
        mysql -uroot -p
            grant all privileges on *.* to 'notebook'@'%';
            ctrl+d
        mysql -u notebook -p
            create database if not exists notebook default charset utf8mb4;
    docker-compose exec app php artisan migrate
    docker-compose exec app php artisan db:seed --class=DatabaseSeeder


if you update Dockerfile(upd php version), run
docker-compose stop
docker-compose build --no-cache --pull app
docker-compose up -d
